\subsection{Đánh giá kết quả}

\textbf{Nhận xét chính:}

\begin{itemize}
    \item \textbf{FairDen đạt Balance cao nhất hoặc cạnh tranh} trên hầu hết các tập dữ liệu:
    \begin{itemize}
        \item Adult (gender): FairDen đạt 0.86, cao hơn FairSC (0.40) và FairSC(N) (0.49)
        \item Adult (race): FairDen đạt 0.83, vượt trội so với các phương pháp khác
        \item Bank: FairDen đạt 0.98, gần như hoàn hảo
        \item Communities: FairDen đạt 0.92, chỉ sau Fairlet (MCF) với 0.99
    \end{itemize}
    
    \item \textbf{DCSI của FairDen cao nhất} (sau DBSCAN) cho hầu hết các dataset, cho thấy FairDen bảo toàn cấu trúc density-connected tốt hơn các phương pháp khác.
    
    \item \textbf{Fairlet và Scalable} chỉ áp dụng được cho thuộc tính nhạy cảm nhị phân. Khi áp dụng được, chúng thường đạt Balance rất cao (0.95-0.99) nhưng DCSI rất thấp (gần 0).
    
    \item \textbf{COMPAS là tập dữ liệu khó nhất}: FairDen chỉ đạt Balance 0.59, thấp hơn Scalable (0.86) do thuộc tính race có 4 giá trị và sự mất cân bằng lớn giữa các nhóm.
    
    \item \textbf{Student Performance}: FairDen đạt Balance 0.93 với DCSI 0.14, cho thấy hiệu quả tốt trên tập dữ liệu giáo dục.
\end{itemize}

\textbf{So sánh với kết quả của tác giả:}

Kết quả thực nghiệm của nhóm tương đồng với xu hướng trong bài báo gốc:
\begin{itemize}
    \item FairDen duy trì vị trí dẫn đầu về Balance trên đa số datasets
    \item Xu hướng DCSI cao cho FairDen so với các phương pháp fair khác được tái hiện
    \item Các phương pháp Fairlet và Scalable tiếp tục cho Balance cao nhưng DCSI thấp
    \item Trong bảng kết quả, các giá trị \hl{highlight} cho thấy Balance của DBSCAN trong thực nghiệm của nhóm (0.71--0.73) cao hơn đáng kể so với giá trị 0.01 trong bài báo gốc.
\end{itemize}
